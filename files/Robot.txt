#Вывод данных на экран
print("Hello World") #Строка

#Обьявление переменной
name = "Иван"/65454543 #Сначала идет имя, потом оператор присваивания - знак =, а потом идет значение переменной
print(name)

В качестве имен для переменных можно использовать:

# Ввод данных
txt = input('Введите текст:') #по умолчанию данные имеют тип string
print(txt)


# Основные типы данных в Python
a = 'Привет' #Строка - string
b = 152 # целое число - intenger
c = 21.3 # число с плавающей точкой - float
d = True # логическое значение - boolean (имеет только два значения True - истина, False - ложь)


# Матетматические операции
print(2 + 5) # сложение
print(2 * 5) # умножение
print(5 - 3) # вычитание
print(8 / 2) # деление
print() # если не указанны параметры, выдается пустая строка
print(9 // 4) # целочисленное деление, возвращается целое число
print(8 // 3) # окгруление берется в меньшую сторону
print(9 / 4) # обыкновенное деление
print(10 % 3) #остаток от деления двух целых чисел
print(3 ** 4) #возведение в степень


#Функции преобразования типов данных
x = input('Введите данные:')
z = int(x) # int - преобразует данные в целочисленные формат
y = str(x) # str - преобразует данные в строку
t = float(x) #float - преобразует данные в число с плавающей точкой
print(type(z), type(y), type(t)) #type - позволяет узнать к какому типу данных относится переменная


# Операторы сравнения
print(2 > 1) # больше
print(3 < 5) # меньше
print(3 == 3) # равенство
print(12 >= 15) # больше или равно
print(12 <= 15) # меньше или равно
print(8 != 5 ) # не равно
